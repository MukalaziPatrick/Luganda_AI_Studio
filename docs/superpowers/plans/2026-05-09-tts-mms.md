> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a `POST /api/v1/tts` endpoint that accepts Luganda text and returns a WAV audio stream using Meta's `facebook/mms-tts-lug` model. Add a speaker button to `translate.html` and `teach.html` with a first-use toast notification.

**Architecture:** A new lazy-loaded TTS service mirrors the pattern of `nllb_service.py`. A new route returns a `StreamingResponse`. The frontend calls the endpoint and plays audio via the HTML5 `<audio>` API. No changes to the translation pipeline.

**Tech Stack:** Python 3.10+, `transformers` (VitsModel + VitsTokenizer), `scipy` for WAV encoding, FastAPI `StreamingResponse`, vanilla JS `fetch` + `Audio` API

# backend/services/tts/mms_tts_service.py

"""
Meta MMS TTS service for Luganda.

Model: facebook/mms-tts-lug
- Luganda-specific voice (real language, not a generic Latin voice)
- CPU-capable, no VRAM required
- ~few hundred MB download
- Lazy-loaded on first request (same pattern as nllb_service.py)

First call: 5–15 s (model loading)
Subsequent: 1–2 s on CPU
"""

import io
import logging
from typing import Optional

import numpy as np
import torch

logger = logging.getLogger(__name__)

MODEL_NAME = "facebook/mms-tts-lug"


class MMSTTSService:
    """Lazy-loaded wrapper around facebook/mms-tts-lug."""

    def __init__(self) -> None:
        self._model = None
        self._tokenizer = None

    def _load(self) -> None:
        if self._model is not None:
            return
        logger.info(f"[MMS-TTS] Loading model {MODEL_NAME} — this takes ~10s on first run")
        from transformers import VitsModel, VitsTokenizer
        self._tokenizer = VitsTokenizer.from_pretrained(MODEL_NAME)
        self._model = VitsModel.from_pretrained(MODEL_NAME)
        self._model.eval()
        logger.info("[MMS-TTS] Model loaded")

    def synthesize(self, text: str) -> Optional[bytes]:
        """
        Synthesize Luganda text into WAV bytes.
        Returns None if synthesis fails.
        """
        try:
            self._load()

            inputs = self._tokenizer(text, return_tensors="pt")

            with torch.no_grad():
                output = self._model(**inputs).waveform

            # output shape: (1, samples) — squeeze to 1D numpy array
            waveform = output.squeeze().cpu().numpy().astype(np.float32)
            sample_rate = self._model.config.sampling_rate

            return _to_wav_bytes(waveform, sample_rate)

        except Exception as e:
            logger.error(f"[MMS-TTS] Synthesis failed: {e}", exc_info=True)
            return None


def _to_wav_bytes(waveform: np.ndarray, sample_rate: int) -> bytes:
    """Convert a float32 numpy waveform array to WAV bytes."""
    import scipy.io.wavfile as wavfile

    # scipy expects int16 for WAV
    pcm = (waveform * 32767).astype(np.int16)
    buf = io.BytesIO()
    wavfile.write(buf, sample_rate, pcm)
    return buf.getvalue()


mms_tts_service = MMSTTSService()
```

# backend/api/routes/tts.py

import logging
from fastapi import APIRouter, HTTPException
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
import io

from backend.services.tts.mms_tts_service import mms_tts_service

logger = logging.getLogger(__name__)
router = APIRouter()


class TTSRequest(BaseModel):
    text: str
    lang: str = "lug"


@router.post("")
async def text_to_speech(request: TTSRequest):
    """
    Synthesize Luganda text into a WAV audio stream.

    Returns: audio/wav stream
    Errors: 400 if text is empty, 503 if model fails
    """
    text = request.text.strip()
    if not text:
        raise HTTPException(status_code=400, detail="text must not be empty")
    if len(text) > 500:
        raise HTTPException(status_code=400, detail="text must be 500 characters or fewer")

    wav_bytes = mms_tts_service.synthesize(text)

    if wav_bytes is None:
        raise HTTPException(status_code=503, detail="TTS synthesis failed. Try again.")

    return StreamingResponse(
        io.BytesIO(wav_bytes),
        media_type="audio/wav",
        headers={"Content-Disposition": "inline; filename=tts.wav"},
    )
```

- [ ] **Step 3: Add the TTS JS to `translate.html`**

At the bottom of `translate.html`, before the closing `</body>` tag, add:

```html
<script>
(function () {
  const btn = document.getElementById('ttsBtnTranslate');
  if (!btn) return;

  let lastLugandaText = '';

  // Called by existing translation result handler — expose via window
  window.setTTSText = function (text) {
    lastLugandaText = text || '';
    btn.style.display = lastLugandaText ? 'inline-block' : 'none';
  };

  btn.addEventListener('click', function () {
    if (!lastLugandaText) return;
    playTTS(lastLugandaText, btn);
  });

  function showFirstUseToast() {
    if (sessionStorage.getItem('tts_toast_shown')) return;
    sessionStorage.setItem('tts_toast_shown', '1');

    const toast = document.createElement('div');
    toast.className = 'tts-toast';
    toast.textContent = 'Loading Luganda voice for the first time — this takes ~10 seconds. It\'ll be instant after that.';
    document.body.appendChild(toast);

    setTimeout(function () {
      toast.classList.add('hiding');
      setTimeout(function () { toast.remove(); }, 350);
    }, 6000);
  }

  function playTTS(text, button) {
    button.disabled = true;
    button.textContent = '⏳';
    showFirstUseToast();

    fetch('/api/v1/tts', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ text: text, lang: 'lug' }),
    })
      .then(function (res) {
        if (!res.ok) throw new Error('TTS request failed: ' + res.status);
        return res.blob();
      })
      .then(function (blob) {
        const url = URL.createObjectURL(blob);
        const audio = new Audio(url);
        button.textContent = '🔊';
        audio.play();
        audio.onended = function () {
          button.disabled = false;
          URL.revokeObjectURL(url);
        };
      })
      .catch(function (err) {
        console.error('TTS error:', err);
        button.textContent = '🔊';
        button.disabled = false;
      });
  }
})();
</script>
```
